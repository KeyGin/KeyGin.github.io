A. LeetCode 8. String to Integer (atoi)

using namespace std;

class Solution {
public:
    int myAtoi(string str) {
        int first_is_digital = -1;
        int sign = 1;
        int temp = 0;
        int digital;
        int low_max = INT32_MAX / 10;
        int low_min = INT32_MIN / 10;
        char ch;
        for (int i = 0; i < str.size(); i++) {
            ch = str[i];
            if (ch > 57 || ch < 48) {
                if (first_is_digital < 0) {
                    if (ch == 32) {
                        continue;
                    } else if (ch == 45) {
                        sign = -1;
                        first_is_digital = 0;
                    } else if (ch == 43) {
                        first_is_digital = 0;
                    } else {
                        return temp;
                    }
                } else {
                    return temp;
                }
                digital = 0;
            } else {
                first_is_digital = 1;
                digital = ch - 48;
                digital = digital * sign;
            }
            if (temp > low_max || (temp == low_max && digital > 7)) {
                return INT32_MAX;
            }
            if (temp < low_min || (temp == low_min && digital < -8)) {
                return INT32_MIN;
            }
            temp = temp * 10 + digital;
        }
        return temp;
    }
};
